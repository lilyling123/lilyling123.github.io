建立数据库

数据库的表包括，stus(sid,sname)，teachers(tid,tname)，lessons(lid,lname,tid)，chose(sid,lid)，card(sid,num)
其中每个学生只有一个card，每个老师可能会上两门课以上，每个学生可能选择若干门课，每门课可能有若干学生选择
这个是一对一、一对多、多对多共同的数据库

编写Mapper

注意：使用了association将card和student对象连接在一起，同时property表示StuCard的字段，cid表示相应的列名，javaType表示对应的对象，然后将获得的数据注入到student对象中，同时记得在SQLConfigMapper的最后面追加这个Mapper的全路径

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--
    namespace：命名空间，它的作用就是对SQL进行分类化管理，可以理解为SQL隔离
    注意：使用mapper代理开发时，namespace有特殊且重要的作用
 -->
<mapper namespace="stuCard">
    <!--一对一关系查询-->
    <select id="findStudentWithCard" parameterType="java.lang.Integer" resultMap="StuCardMap">
        SELECT * FROM stus,card WHERE stus.sid=#{id} AND stus.sid=card.cid
    </select>

    <resultMap id="StuCardMap" type="com.lily.po.StuCard">
        <id column="cid" property="cid"></id>
        <result column="num" property="num"></result>
        <association property="student" column="cid" javaType="com.lily.po.Student">
            <id column="sid" property="sid"></id>
            <result column="sname" property="sname"></result>
        </association>
    </resultMap>
</mapper>

业务：查找某个老师上的课，是一对多

环境沿用上面的

编写实体类

编写mapper

注：此处使用了collection标签，property代表TeacherAndLesson中的属性，Lesson代表具体类型，tid代表约束

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--
    namespace：命名空间，它的作用就是对SQL进行分类化管理，可以理解为SQL隔离
    注意：使用mapper代理开发时，namespace有特殊且重要的作用
 -->
<mapper namespace="teacherMap">
    <!--一对多关系查询-->
    <select id="findTeacherWithLesson" parameterType="java.lang.Integer" resultMap="TeacherLessonMap">
        SELECT * FROM teachers,lessons WHERE teachers.tid=#{id} AND lessons.tid=teachers.tid
    </select>
    <resultMap id="TeacherLessonMap" type="com.lily.po.TeacherAndLesson">
        <id column="tid" property="tid"></id>
        <result column="tname" property="tname"></result>
        <collection property="lessons" ofType="com.lily.po.Lesson" column="tid">
            <id column="lid" property="lid"></id>
            <result column="lname" property="lname"></result>
            <result column="tid" property="tid"></result>
        </collection>
    </resultMap>
</mapper>

测试

public void findTeacherWithLesson() {
    SqlSession sqlSession = sqlSessionFactory.openSession();

    TeacherAndLesson s = sqlSession.selectOne("teacherMap.findTeacherWithLesson", 101);
    for (Lesson l : s.getLessons())
        System.out.println(l.getLname());
}

MyBatis实现的多对多的关系映射查询

业务：老师打印点名表，一个老师上多门课，每门课有多名学生，因此是多对多

环境沿用上面的

编写实体类

编写mapper

注：此处使用了嵌套的collection标签进行注入

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--
    namespace：命名空间，它的作用就是对SQL进行分类化管理，可以理解为SQL隔离
    注意：使用mapper代理开发时，namespace有特殊且重要的作用
 -->
<mapper namespace="teacherCallTable">
    <!--一对多关系查询-->
    <select id="findCallTableOfTeacher" parameterType="java.lang.Integer" resultMap="CallTableOfTeacherMap">
        SELECT * FROM teachers,lessons,stus,chose WHERE teachers.tid=#{id} AND lessons.tid=teachers.tid AND chose.lid=lessons.lid AND stus.sid=chose.sid
    </select>
    <resultMap id="CallTableOfTeacherMap" type="com.lily.po.CallTable">
        <id column="tid" property="tid"></id>
        <result column="tname" property="tname"></result>
        <collection property="lessons" ofType="com.lily.po.Lesson" column="tid">
            <id column="lid" property="lid"></id>
            <result column="lname" property="lname"></result>
            <result column="tid" property="tid"></result>
            <collection property="students" ofType="com.lily.po.Student" column="sid">
                <id column="sid" property="sid"></id>
                <result column="sname" property="sname"></result>
            </collection>
        </collection>
    </resultMap>
</mapper>

测试

public void findCallTableOfTeacher() {
    SqlSession sqlSession = sqlSessionFactory.openSession();

    List<CallTable> list = sqlSession.selectList("teacherCallTable.findCallTableOfTeacher", 101);
    for (CallTable c : list) {
        List<Lesson> lessons = c.getLessons();
        for(Lesson l : lessons){
            System.out.println(l.getStudents());
        }
    }
}
